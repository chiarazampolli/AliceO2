48 void Detector::createMaterials()
49 {
50  // implement materials here
51  // Definition of available HMPID materials
52  // Arguments: none
53  // Returns: none
54 
55  // clm update material definition later on from Antonello
56  // Ported to O2 (24.4.2018) -- taken from AliRoot AliHMPIDv3
57 
58  // data from PDG booklet 2002 density [gr/cm^3] rad len [cm] abs len [cm]
59  float aAir[4] = { 12, 14, 16, 36 }, zAir[4] = { 6, 7, 8, 18 }, wAir[4] = { 0.000124, 0.755267, 0.231781, 0.012827 },
60  dAir = 0.00120479;
61  Int_t nAir = 4; // mixture 0.9999999
62  float aC6F14[2] = { 12.01, 18.99 }, zC6F14[2] = { 6, 9 }, wC6F14[2] = { 6, 14 }, dC6F14 = 1.68;
63  Int_t nC6F14 = -2;
64  float aSiO2[2] = { 28.09, 15.99 }, zSiO2[2] = { 14, 8 }, wSiO2[2] = { 1, 2 }, dSiO2 = 2.64;
65  Int_t nSiO2 = -2;
66  float aCH4[2] = { 12.01, 1.01 }, zCH4[2] = { 6, 1 }, wCH4[2] = { 1, 4 }, dCH4 = 7.17e-4;
67  Int_t nCH4 = -2;
68 
69  float aRoha = 12.01, zRoha = 6, dRoha = 0.10, radRoha = 18.80,
70  absRoha = 86.3 / dRoha; // special material- quasi quartz
71  float aCu = 63.55, zCu = 29, dCu = 8.96, radCu = 1.43, absCu = 134.9 / dCu;
72  float aW = 183.84, zW = 74, dW = 19.30, radW = 0.35, absW = 185.0 / dW;
73  float aAl = 26.98, zAl = 13, dAl = 2.70, radAl = 8.90, absAl = 106.4 / dAl;
74  float aAr = 39.94, zAr = 18, dAr = 1.396e-3, radAr = 14.0, absAr = 117.2 / dAr;
75 
76  Int_t matId = 0; // tmp material id number
77  Int_t unsens = 0, sens = 1; // sensitive or unsensitive medium
78  Int_t itgfld;
79  float maxfld;
80  o2::Base::Detector::initFieldTrackingParams(itgfld, maxfld);
81 
82  float tmaxfd = -10.0; // max deflection angle due to magnetic field in one step
83  float deemax = -0.2; // max fractional energy loss in one step
84  float stemax = -0.1; // max step allowed [cm]
85  float epsil = 0.001; // abs tracking precision [cm]
86  float stmin = -0.001; // min step size [cm] in continius process transport, negative value: choose it automatically
87 
88  // PCB copmposed mainly by G10 (Si,C,H,O) -> CsI is negligible (<500nm thick)
89  // So what is called CsI has the optical properties of CsI, but the composition of G-10 (for delta elec, etc
90  // production...)
91 
92  float aG10[4] = { 28.09, 12.01, 1.01, 16.00 };
93  float zG10[4] = { 14., 6., 1., 8. };
94  float wG10[4] = { 0.129060, 0.515016, 0.061873, 0.294050 };
95  float dG10 = 1.7;
96  Int_t nG10 = 4;
97 
98  Mixture(++matId, "Air", aAir, zAir, dAir, nAir, wAir);
99  Medium(kAir, "Air", matId, unsens, itgfld, maxfld, tmaxfd, stemax, deemax, epsil, stmin);
100 
101  Mixture(++matId, "C6F14", aC6F14, zC6F14, dC6F14, nC6F14, wC6F14);
102  Medium(kC6F14, "C6F14", matId, unsens, itgfld, maxfld, tmaxfd, stemax, deemax, epsil, stmin);
103 
104  Mixture(++matId, "SiO2", aSiO2, zSiO2, dSiO2, nSiO2, wSiO2);
105  Medium(kSiO2, "SiO2", matId, unsens, itgfld, maxfld, tmaxfd, stemax, deemax, epsil, stmin);
106 
107  Mixture(++matId, "CH4", aCH4, zCH4, dCH4, nCH4, wCH4);
108  Medium(kCH4, "CH4", matId, unsens, itgfld, maxfld, tmaxfd, stemax, deemax, epsil, stmin);
109 
110  Mixture(++matId, "CsI+PCB", aG10, zG10, dG10, nG10, wG10);
111  Medium(kCsI, "CsI", matId, sens, itgfld, maxfld, tmaxfd, stemax, deemax, epsil, stmin); // sensitive
112 
113  Mixture(++matId, "Neo", aSiO2, zSiO2, dSiO2, nSiO2, wSiO2);
114  Medium(kNeo, "Neo", matId, unsens, itgfld, maxfld, tmaxfd, stemax, deemax, epsil, stmin); // clm neoceram
115 
116  Material(++matId, "Roha", aRoha, zRoha, dRoha, radRoha, absRoha);
117  Medium(kRoha, "Roha", matId, unsens, itgfld, maxfld, tmaxfd, stemax, deemax, epsil, stmin); // Roha->honeycomb
118

120  Medium(kCu, "Cu", matId, unsens, itgfld, maxfld, tmaxfd, stemax, deemax, epsil, stmin);
