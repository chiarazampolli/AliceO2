63  // In case no extra argument is provided and the passed type is trivially
64  // copyable and non polymorphic, the most likely wanted behavior is to create
65  // a message with that type, and so we do.
66  template <typename T>
67  typename std::enable_if<is_messageable<T>::value == true, T&>::type
68  make(const Output& spec)
69  {
70  DataChunk chunk = newChunk(spec, sizeof(T));
71  return *reinterpret_cast<T*>(chunk.data);
72  }

179  template <typename W>
180  typename std::enable_if<is_specialization<W, ROOTSerialized>::value == true, void>::type
181  snapshot(const Output& spec, W wrapper)
182  {
183  using T = typename W::wrapped_type;
184  static_assert(std::is_same<typename W::hint_type, const char>::value || //
185  std::is_same<typename W::hint_type, TClass>::value || //
186  std::is_void<typename W::hint_type>::value, //
187  "class hint must be of type TClass or const char");
188 
189  FairMQMessagePtr payloadMessage(mDevice->NewMessage());
190  const TClass* cl = nullptr;
191  if (wrapper.getHint() == nullptr) {
192  // get TClass info by wrapped type
193  cl = TClass::GetClass(typeid(T));
194  } else if (std::is_same<typename W::hint_type, TClass>::value) {
195  // the class info has been passed directly
196  cl = reinterpret_cast<const TClass*>(wrapper.getHint());
197  } else if (std::is_same<typename W::hint_type, const char>::value) {
198  // get TClass info by optional name
199  cl = TClass::GetClass(reinterpret_cast<const char*>(wrapper.getHint()));
200  }
201  if (has_root_dictionary<T>::value == false && cl == nullptr) {
202  std::string msg("ROOT serialization not supported, dictionary not found for type ");
203  if (std::is_same<typename W::hint_type, const char>::value) {
204  msg += reinterpret_cast<const char*>(wrapper.getHint());
205  } else {
206  msg += typeid(T).name();
207  }
208  throw std::runtime_error(msg);
209  }
210  mDevice->Serialize<TMessageSerializer>(*payloadMessage, &wrapper(), cl);
211  addPartToContext(std::move(payloadMessage), spec, o2::header::gSerializationMethodROOT);
212  }

266  for (auto const & pointer : v) {
