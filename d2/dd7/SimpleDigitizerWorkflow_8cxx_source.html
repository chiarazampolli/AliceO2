30 // extract num TPC lanes, a lane is a streaming line of processors (digitizer-clusterizer-etc)

43 void extractTPCSectors(std::vector<int>& sectors)
44 {
45  if (const char* f = std::getenv("DPL_TPC_SECTORS")) {
46  // we expect them to be , separated
47  std::stringstream ss(f);
48  std::vector<std::string> stringtokens;
49  while (ss.good()) {
50  std::string substr;
51  getline(ss, substr, ',');
52  stringtokens.push_back(substr);
53  }
54  // now try to convert each token to int
55  for (auto& token : stringtokens) {
56  try {
57  auto s = std::stoi(token);
58  sectors.emplace_back(s);
59  } catch (std::invalid_argument e) {
60  }
61  }
62  } else {
63  // all sectors otherwise by default
64  for (int s = 0; s < o2::TPC::Sector::MAXSECTOR; ++s) {
65  sectors.emplace_back(s);
66  }
67  }
68 }
